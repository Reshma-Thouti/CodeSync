class Solution:
    def countVowelStrings(self, n: int) -> int:
        #normal method
        # m,j=1,1
        # for i in range(5,5+n):
        #     m=(m*i)//j
        #     j+=1
        # return m

        #DP
        dp=[1]*5
        for _ in range(n-1):
            for i in range(3,-1,-1):
                dp[i]+=dp[i+1]
        return sum(dp)
